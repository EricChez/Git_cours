Le spread operator (ou syntaxe de décomposition en français) existe aussi en PHP.

The spread operator used with arrays

var x = [1, 2, 3]
var y = [4, 5, 6]
// copy an array
var xx = [...x]
console.log(xx)

// concatenation of arrays

var array_one = ['dog', 'cat', 'otter']
var array_two = ['lion', 'googleplex', 123]
var array_three = [...array_one, ...array_two]
console.log(array_three)

// add an element

var array_four = ['requin', ... array_one]
console.log(array_four)

// spread a string in an array

var string = "geek"
console.log(array_string)

The spread operator used with objects

var objectOne = {
id: 3,
name:'John James'
}

var objectTwo = {
age: 47,
country: 'UK'
}

var person = {...objectOne, ...objectTwo}
console.log(person)

objectOne.address = "14 Serpent Street"
console.log(objectOne)
objectOne.otherPerson = {...objectOne}
console.log(objectOne)
delete objectOne.otherPerson
console.log(objectOne)

// Nullish coalescing operator (operateur de coalescence null)

var text = null ?? 'Tata Yoyo'
console.log(text)

var img = null
var imgUrl = img ?? "https://exemple.com"
console.log(imgUrl)

var koko = "noix de coco"
var zazou = "noix de cajou"
var kokotest = koko ?? zazou
console.log(kokotest)

// Nullish coalescing assignment (affectation après coalescence des nuls)

var objThree = {
id: 74,
value: 47
}

objThree.id ??= 12 // returns 74

objThree.goal ??= "nothing" // creates a new key-value par = goal: "nothing"

console.log(objThree)
